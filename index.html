function castFusion() { let fusionType = getFusionType(); // Define the variants for the fusions const variants = [ { type: 'fire', effect: 'burns the enemy', variantModifier: 1.5 }, { type: 'water', effect: 'freezes the enemy', variantModifier: 1.5 }, { type: 'earth', effect: 'stuns the enemy', variantModifier: 1.5 }, ]; // Add random variants for each fusion ultimate for (let i = 0; i < 3; i++) { let randomVariant = variants[sRand(0, variants.length - 1)]; console.log(`Fusion of ${fusionType} with ${randomVariant.type} - Effect: ${randomVariant.effect} (Modifier: ${randomVariant.variantModifier})`); } } function sRand(min, max) { return Math.floor(Math.random() * (max - min + 1)) + min; }
